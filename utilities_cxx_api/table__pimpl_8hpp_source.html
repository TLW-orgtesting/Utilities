384 template<typename ContainerType>
385 std::string row_spacer(const ContainerType& widths,
386  const std::array<std::string, 3>& borders,
387  char underline, char intersect) {
388  const auto n_cols = widths.size();
389 
390  // Determine the termination string for the right-side of a column
391  const auto npad = borders[1].size() - 1;
392  if(npad % 2 != 0)
393  throw std::runtime_error("Padding on middle border must be symmetric");
394 
395  const auto rhs = std::string(npad / 2, underline) + intersect +
396  std::string(npad / 2, underline);
397 
398  // Assemble the row spacer
399  std::stringstream rv;
400 
401  // The 1st intersect and any extra characters to account for lborder_
402  rv << intersect << std::string(borders[0].size() - 1, underline);
403 
404  // The underline of each column except the last
405  for(auto i = 0; i < n_cols - 1; ++i) {
406  rv << std::string(widths[i], underline) + rhs;
407  }
408 
409  // The last column
410  rv << std::string(widths[n_cols - 1], underline);
411  rv << std::string(borders[2].size() - 1, underline) << intersect;
412 
413  return rv.str();
414 }

416 template<typename ContainerType, typename DataType>
417 std::string print_row(const ContainerType& widths, const DataType& row_data,
418  const std::array<std::string, 3>& borders) {
419  std::stringstream ss;
420 
421  // Determine the height of this row
422  auto l = [](auto i, auto j) { return i.size() < j.size(); };
423  auto height = std::max_element(row_data.begin(), row_data.end(), l)->size();
424 
425  const auto ncols = widths.size();
426 
427  for(auto line_i = 0; line_i < height; ++line_i) {
428  ss << borders[0];
429  for(auto col_i = 0; col_i < ncols; ++col_i) {
430  const auto width_i = widths[col_i];
431 
432  if(row_data[col_i].size() <= line_i) { // No more lines, print blank
433  ss << std::string(width_i, ' ');
434  } else {
435  const auto& line = row_data[col_i][line_i];
436 
437  if(line.size() > width_i)
438  throw std::runtime_error(
439  "Line is larger than column width");
440 
441  ss << line << std::string(width_i - line.size(), ' ');
442  }
443  ss << (col_i != ncols - 1 ? borders[1] : borders[2]);
444  }
445  ss << std::endl;
446  }
447  return ss.str();
448 }

538  size_type col_i) const {
539  if(m_widths_.at(col_i) > 0) // Set by user
540  return m_widths_[col_i];
541 
542  size_type w = 0;
543  for(std::size_t row_i = 0; row_i < nrows(); ++row_i) {
544  // Split contents into sentences

554 inline typename TablePIMPL::size_vec TablePIMPL::space_out_cols() const {
555  const auto n = ncols();
556  if(n == 0) return size_vec{};
557 
558  // How many characters do we loose to borders?
559  const auto nborders = m_borders_[0].size() +
560  (n - 1) * m_borders_[1].size() + m_borders_[2].size();
561 
562  // How many characters did the user fix?
563  auto user_chars = std::accumulate(m_widths_.begin(), m_widths_.end(), 0);
564 
565  // See if we have enough characters to honor user's requests
566  if(user_chars + nborders > m_max_w_)
567  throw std::runtime_error("Not enough char's to meet user's request");
568 
569  // We assume the message is printable, i.e., we can at least do one word per
570  // line
571 
572  // Which columns do we need to figure out sizes for?
573  size_vec auto_cols;
574  for(size_type i = 0; i < n; ++i)
575  if(m_widths_[i] == 0) auto_cols.push_back(i);
576 
577  // User specified all of them and they're printable, i.e., we're done
578  if(auto_cols.empty()) return m_widths_;
579 
580  // How many chars do we have left to work with
581  const auto free_chars = m_max_w_ - user_chars - nborders;
582 
583  // Our plan is to give each of the remaining columns their width in char's
584  // If that doesn't work we just give each column the same number of char's
585 
586  size_vec rv(m_widths_); // Our return value
587 
588  // How many char's are the column's widths?
589  auto nrequest = 0;
590  for(auto x : auto_cols) nrequest += width_of_col(x);
591 
592  const auto nauto = auto_cols.size();
593 
594  // What's the average number of characters?
595  auto extra_chars = free_chars % nauto;
596  auto avg_free_chars = (free_chars - extra_chars) / nauto;
597 
598  // Can we just give them their widths?
599  bool is_good = nrequest <= free_chars;
600 
601  for(auto x : auto_cols) rv[x] = is_good ? width_of_col(x) : avg_free_chars;
602 
603  return rv;
604 }
