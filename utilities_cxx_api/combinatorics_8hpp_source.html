258 template<typename container_type>
259 std::deque<std::size_t> permutation_to_fns(const container_type& perm,
260  const container_type& original) {
261  // Recursion end
262  if(perm.size() == 0) return std::deque<std::size_t>({});
263  if(perm.size() == 1) return std::deque<std::size_t>({0});
264 
265  // Strip off 1st element and put remaining elements in new container
266  auto perm_itr = perm.begin();
267  auto ei = *perm_itr;
268  container_type sub_perm(++perm_itr, perm.end());
269 
270  // Get digit in FNS and fill new_orig with all elements aside from it
271  std::size_t digit = 0;
272  container_type new_orig;
273  auto orig_itr = original.begin();
274  for(std::size_t i = 0; i < original.size(); ++i) {
275  auto ej = *orig_itr++;
276  if(ej == ei) {
277  digit = i;
278  while(orig_itr != original.end()) {
279  new_orig.insert(new_orig.end(), *orig_itr++);
280  }
281  break;
282  }
283  new_orig.insert(new_orig.end(), ej);
284  }
285  auto rv = permutation_to_fns(sub_perm, new_orig);
286  rv.push_front(digit);
287  return rv;
288 }

292  const container_type& original) {
293  container_type temp(original);
294  container_type rv;
295  for(auto x : fns) {
296  auto ei_itr = temp.begin();
297  container_type temp2;
298  for(std::size_t i = 0; i < temp.size(); ++i) {
299  if(i != x)
300  temp2.insert(temp2.end(), *ei_itr);
301  else
302  rv.insert(rv.end(), *ei_itr);
303  ++ei_itr;

328 std::deque<std::size_t> decimal_to_fns(std::size_t D,
329  const container_type& orig) {
330  container_type new_orig(orig.begin(), orig.end());
331  std::sort(new_orig.begin(), new_orig.end());
332  std::deque<std::size_t> rv;
333 
334  while(new_orig.begin() !=
335  new_orig.end()) // We pop an element out per iteration
336  {
337  auto curr_guess = new_orig.begin();
338  std::size_t total_perms = 0;
339  std::size_t digit_i = 0;
340  while(curr_guess != new_orig.end()) {
341  auto ei = *curr_guess;
342  // New container w/o element i
343  container_type temp(new_orig.begin(), curr_guess++);
344  temp.insert(temp.end(), curr_guess, new_orig.end());
345 
346  const auto P_ei = n_permutations(temp);
347  total_perms += P_ei;
348  if(D < total_perms) {
349  rv.push_back(digit_i);
350  new_orig.swap(temp);
351  D = D - (total_perms - P_ei);
352  break;
353  }
354  // We know the element isn't ei, but ei may still be in temp...
355  auto n_ei = std::count(temp.begin(), temp.end(), ei);
356  digit_i += 1 + n_ei;
357  std::advance(curr_guess, n_ei);
358  }
359  }
360 
361  return rv;
362 }
